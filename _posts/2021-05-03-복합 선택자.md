# 일치 선택자(Basic Combinator)
E와 F를 동시에 만족하는 요소 선택

{% highlight html %}

<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title>Insert title here</title>
    <style>
      span.orange{
        color:red;
      }
    </style>
	</head>
	<body>
        <div>
          <ul>
            <li>사과</li>
            <li>딸기</li>
            <li class="orange">오렌지</li> <!-- 선택 -->
          </ul>
          <div>당근</div>
          <p>토마토</p>
          <span classs="orange">오렌지</span> <!-- 선택 -->
        </div>
	</body>
</html>

{% endhighlight %}


# 자식 선택자(Child Combinator)
E의 자식 요소 F를 선택

{% highlight html %}

<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title>Insert title here</title>
    <style>
      ul > .orange{
        color:red;
      }

{% endhighlight %}


# 후손(하위) 선택자(Descendent Combinator)
E의 후손(하위)요소 F를 선택

{% highlight html %}

<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title>Insert title here</title>
    <style>
      div .orange{
        color:red;
      }
    </style>
	</head>
	<body>
        <div>
          <ul>
            <li>사과</li>

	</body>
</html>

{% endhighlight %}


# 인접 형제 선택자(Adjacent Sibling Combinator)
E의 다음 형제 요소 F 하나만 선택

{% highlight html %}

<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title>Insert title here</title>
    <style>
      .orange + li{
        color:red;
      }
    </style>
	</head>
	<body>
        <div>
          <ul>
            <li>사과</li>
            <li>딸기</li>
            <li class="orange">오렌지</li>
            <li>당근</li> <!-- 선택 -->
            <li>토마토</li>
           </ul>
        </div>
	</body>
</html>

{% endhighlight %}
